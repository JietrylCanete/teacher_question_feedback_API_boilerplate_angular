  const lowerAnswer = answerText.toLowerCase();
  const lowerQuestion = questionText.toLowerCase();

  const keywords = lowerQuestion
    .split(/\W+/)
    .filter(word => word.length > 4); // only medium/long words

  const matches = keywords.filter(kw => lowerAnswer.includes(kw));

  const isRelevant = matches.length > 0 || answerText.length > 20;

  // Generate a realistic-sounding feedback

    const templates = [
      "The answer addresses the topic of the question.",
      "This is relevant and shows understanding of the question.",

    ];

      "The answer does not address the question clearly.",
      "This response is off-topic or incomplete.",
      "The student’s answer is not relevant to the question."
    ];

  }

  };

// ai.service.js — offline relevance checker
function reviewAnswer(questionText, answerText) {
  questionText = questionText.toLowerCase();
  answerText = answerText.toLowerCase();

  // Extract keywords from question (simple split)
  const keywords = questionText
    .replace(/[^\w\s]/g, "")
    .split(/\s+/)
    .filter(w => w.length > 3); // skip very short words

  // Count how many keywords appear in answer
  const matchCount = keywords.filter(word => answerText.includes(word)).length;

  // Decide relevance: >=30% keywords present
  const threshold = Math.max(1, Math.floor(keywords.length * 0.3));
  const isRelevant = matchCount >= threshold;

  const feedback = isRelevant
    ? `The answer mentions ${matchCount} key concept(s) from the question.`
    : `The answer misses most key concepts (${matchCount}/${keywords.length} words matched).`;

  return { isRelevant, feedback };
}

module.exports = { reviewAnswer };
